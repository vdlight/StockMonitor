    // Member variables to hold results of each calculation
    public decimal Result1 { get; private set; }
    public decimal Result2 { get; private set; }
    public decimal Result3 { get; private set; }

    public void PerformCalculations()
    {
        // List of tasks to handle each calculation independently
        List<Task<decimal>> tasks = new List<Task<decimal>>
        {
            Task.Run(() => CalculatePart1()),
            Task.Run(() => CalculatePart2()),
            Task.Run(() => CalculatePart3())
        };

        // Wait for all tasks to complete
        Task.WhenAll(tasks).Wait();

        // Assign task results to member variables
        Result1 = tasks[0].Result;
        Result2 = tasks[1].Result;
        Result3 = tasks[2].Result;
    }
